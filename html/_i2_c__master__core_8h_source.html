21 void I2C_Initialize(void)
22 {
23  TWSR = 0; // pas de préscaler
24  TWBR = ((F_CPU/SCL_CLOCK)-16)/2; // Doit être supérieur à 10 pour que les opérations soient stables
25  TWDR = 0xFF; // SDA released
26  TWCR = (0<<TWINT)| // Disable Interupt
27  (0<<TWEA)| // Désactivation du bit d'aquittement
28  (0<<TWSTA)| // Désactivation du bit de condition start
29  (0<<TWSTO)| // Désactivation du bit de condition stop
30  (0<<TWWC)| // Désactivation du drapeau de collision
31  (1<<TWEN)| // Activation de l'interface TWI
32  (0<<TWIE); // Désactivation des interruptions TWI
33 }

35 void I2C_Stop(void)
36 {
37  // Envoi d'une condition STOP
38  TWCR = (1<<TWINT) | (1<<TWEN) | (1<<TWSTO);
39 
40  // Attente jusqu'à ce que la condition STOP soit transmise et que le bus soit relaché
41  while (TWCR & (1<<TWSTO));

44 uint8_t I2C_Start(const uint8_t address)
45 {
46  // Envoi de la condition START
47  TWCR = (1<<TWINT) | (1<<TWSTA) | (1<<TWEN);
48 
49  // Attente de la fin de transmission
50  while (!(TWCR & (1<<TWINT)));
51 
52  // Vérification du registre de status TWI
53  if ( (TW_STATUS != TW_START)
54  && (TW_STATUS != TW_REP_START) )
55  {
56  return 1;
57  }
58 
59  // Envoi de l'adresse du périphérique
60  TWDR = address;
61  TWCR = (1<<TWINT) | (1<<TWEN);
62 
63  // Attente de la réception d'un ACK ou NACK
64  while(!(TWCR & (1<<TWINT)));
65 
66  // Vérification du registre de status TWI
67  if ( (TW_STATUS != TW_MT_SLA_ACK)
68  && (TW_STATUS != TW_MR_SLA_ACK) )
69  {
70  return 1;
71  }
72 
73  return 0;
74 }
