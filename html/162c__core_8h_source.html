51 static void Strobe(void)
52 {
53  // On allume la limière
54  LCD_CONTROL_PORT |= _BV(LCD_CONTROL_EN_PIN);
55  __asm__("NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;");
56  __asm__("NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;");
57  // On éteint la lumière pour que le LCD puisse "réfléchir"
58  LCD_CONTROL_PORT &= ~_BV(LCD_CONTROL_EN_PIN);
59 }

61 static void CheckIfLcdIsBusy(void)
62 {
66  // On positionne le bus de donnée du µC en lecture
67  LCD_DATA_DDR = 0x00;
68 
69  // set R/W et unset RS
70  LCD_CONTROL_PORT &= ~_BV(LCD_CONTROL_RS_PIN); // Turn on Mr LCD's Command Mode (RS off)
71  LCD_CONTROL_PORT |= _BV(LCD_CONTROL_RW_PIN); // Set Mr. LCD to Read (RW on)
72 
73  while (LCD_DATA_PORT >= LCD_MSK_BUSY_FLG)
74  {
75  Strobe();
76  }
77 
78  // On positionne le bus de donnée du µC en écriture
79  LCD_DATA_DDR = 0xFF;
80 }

82 static void SendCommand(const uint8_t command)
83 {
84  CheckIfLcdIsBusy();
85  LCD_DATA_PORT = command;
86  LCD_CONTROL_PORT &= ~(_BV(LCD_CONTROL_RW_PIN)|_BV(LCD_CONTROL_RS_PIN)); // (RS off) (RW off)
87  Strobe();
88  LCD_DATA_PORT = 0x00;
89 }
