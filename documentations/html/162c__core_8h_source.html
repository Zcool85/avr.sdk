14 static void Strobe(void)
15 {
16  // On allume la limière
17  LCD_CONTROL_PORT |= _BV(LCD_CONTROL_EN_PIN);
18  __asm__("NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;");
19  __asm__("NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;NOP;");
20  // On éteint la lumière pour que le LCD puisse "réfléchir"
21  LCD_CONTROL_PORT &= ~_BV(LCD_CONTROL_EN_PIN);
22 }

24 static void CheckIfLcdIsBusy(void)
25 {
26  // On positionne le bus de donnée du µC en lecture
27  LCD_DATA_DIR = 0x00;
28 
29  // set R/W et unset RS
30  LCD_CONTROL_PORT &= ~_BV(LCD_CONTROL_RS_PIN); // Turn on Mr LCD's Command Mode (RS off)
31  LCD_CONTROL_PORT |= _BV(LCD_CONTROL_RW_PIN); // Set Mr. LCD to Read (RW on)
32 
33  while (LCD_DATA_PORT >= LCD_MSK_BUSY_FLG)
34  {
35  Strobe();
36  }
37 
38  // On positionne le bus de donnée du µC en écriture
39  LCD_DATA_DIR = 0xFF;
40 }

42 static void SendCommand(const unsigned char command)
43 {
44  CheckIfLcdIsBusy();
45  LCD_DATA_PORT = command;
46  LCD_CONTROL_PORT &= ~(_BV(LCD_CONTROL_RW_PIN)|_BV(LCD_CONTROL_RS_PIN)); // (RS off) (RW off)
47  Strobe();
48  LCD_DATA_PORT = 0x00;
49 }

51 void LCD_PrintChar(const unsigned char character)
52 {
53  CheckIfLcdIsBusy();
54  LCD_DATA_PORT = character;
55  LCD_CONTROL_PORT &= ~_BV(LCD_CONTROL_RW_PIN); // (RW off)
56  LCD_CONTROL_PORT |= _BV(LCD_CONTROL_RS_PIN); // (RS on)
57  Strobe();
58  _delay_us(46);
59  LCD_DATA_PORT = 0x00;
60 }

70 void LCD_Initialize(void)
71 {
72  LCD_CONTROL_DIR |= _BV(LCD_CONTROL_EN_PIN)|_BV(LCD_CONTROL_RW_PIN)|_BV(LCD_CONTROL_RS_PIN); // (EN on) (RS on) (RW on)
73  LCD_DATA_DIR |= 0xFF; // Port des datas en sortie
74 
75  // wait 10 ms for busy state
76  _delay_ms(10);
77  // Attente de la fin d'initialisation du LCD
78  CheckIfLcdIsBusy();
79 }
