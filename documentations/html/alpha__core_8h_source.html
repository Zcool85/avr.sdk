57  // Configuration Setting Command
58  // -----------------------------
59  // 100 : Command
60  // b1 b0 : Frequency Band (01 => 433 MHz)
61  // eb : Enable low battery detector
62  // et : Enable wake-up timer
63  // ex : Is set id the crystal is active during sleep mode.
64  // x3 x2 x1 x0 : Crystal Load Capacitance (1000 => 12.5 pF)
65  // i2 i1 i0 : Baseband Bandwith (101 => 134 KHz)
66  // dc : Disable the clock Output
67  SPI_SendWord(0b1000100110001011);
68 
69  // Frequency Setting Command
70  // -------------------------
71  // 1010 : Command
72  // f11 f10 f9 f8 f7 f6 f5 f4 f3 f2 f1 f0 : Should be in range of 96 and 3903 (011000100000 => 1568)
73  //
74  // Fo = 10 MHz * (43 + F / 4000) => Fo = 433.92 MHz
75  //
76  // NOTE : Configure Frequency BEFORE starting Synthesizer
77  SPI_SendWord(0b1010011000100000);
78 
79  // Receiver Setting Command
80  // ------------------------
81  // 11000000 : Command
82  // d1 d0 : Select VDI (Valid Data Indicator) signal (11 => VDI Always on)
83  // g1 g0 : LNA Gain (00 => 0 dB relative to max G)
84  // r2 r1 r0 : DRSSI Threshold (000 => -103 dBm)
85  // en : Enable whole receiver chain (wake-up & low battery detector are not affected by this setting)
86  //
87  // RSSIth = RSSIsetth + Glna => RSSIth = -103 + LNA Gain
88  SPI_SendWord(0b1100000011000001);
89 
90  // Wake-Up Timer Command
91  // ---------------------
92  // 111 : Command
93  // r4 r3 r2 r1 r0 : Should be in range of 0 to 23
94  // m7 m6 m5 m4 m3 m2 m1 m0 :
95  //
96  // T = M * 2^R => 0ms
97  //
98  // NOTE : For continual operation, bit et must be cleared and set
99  //SPI_SendWord(0b1110000000000000);
100 
101  // Low Duty-Cycle Command
102  // ----------------------
103  // 11001100 : Command
104  // d6 d5 d4 d3 d2 d1 d0 : (0000111 => 7)
105  // en : Enable low duty cycle mode
106  SPI_SendWord(0b1100110000001110); // Low Duty-Cycle Command : DutyCycle = (D * 2 + 1) / M * 100%; en = 0
107 
108  // Low Battery Detector & Microcontroller Clock Divider Command
109  // ------------------------------------------------------------
110  // 11000010 : Command
111  // d2 d1 d0 : Clock Ouptut Frequency (111 => 10 MHz)
112  // t4 t3 t2 t1 t0 : Threshold voltage
113  //
114  // Vlowbat = 2.2 + T * 0.1 => Vlowbat = 2.2 V
115  SPI_SendWord(0b1100001011100000);
116 
117  // AFC Command
118  // -----------
119  // 11000110 : Command
120  // a1 a0 : Select AFC auto-mode (11 => Keeps independently from VDI)
121  // rl1 rl0 : Range limit (11 => +3-4)
122  // st : When goes high, the actual latest calculated frequency error is stored into the output registers of the AFC block
123  // fi : Enable high accuracy mode. The processing time is about 4 times longer
124  // oe : Enable the output (frequency offset) register
125  // en : Enable calculation of the offset frequency by the AFC circuit (if allows the addition of the content of the output register to the frequency control word of the PPL)
126  SPI_SendWord(0b1100011011110111);
127 
128  // Data Filter Command
129  // -------------------
130  // 11000100 : Command
131  // al : Enable Clock Recovery auto lock control. It means that the CR start in fast mode after locking it automatically switches to slow mode
132  // ml : Enable fast mode
133  // 1 : -
134  // s1 s0 : Type of data Filter (01 => Digital filter)
135  // f2 f1 f0 : DQD threchold (100 => 4)
136  SPI_SendWord(0b1100010011101100);
137 
138  // Data Rate Command
139  // -----------------
140  // 11001000 : Command
141  // cs :
142  // r6 r5 r4 r3 r2 r1 r0 : DataRate (0100011 => 35)
143  //
144  // BaudRate = 10 MHz / 29 / (DataRate + 1) / (1 + cs * 7) => BaudRate = 9578,5440613026819923371647509579 bauds
145  //
146  // Set the Receiver DataRate according the next function : DataRate = (10 MHz / 29 / (1 + cs * 7) / BaudRate) - 1
147  SPI_SendWord(0b1100100000100011);
148 
149  // Output and FIFO mode Command
150  // ----------------------------
151  // 11001110 : Command
152  // f3 f2 f1 f0 : FIFO IT level. The FIFO generate IT when number of the received data bits reaches this level (1000 => 8)
153  // s1 s0 : Select the input of the FIFO fill start condition (10 => VDI & Sync-word)
154  // ff : Enable FIFO fill after synchron word reception. FIFO fill stop when this bit is cleared
155  // fe : Enables the 16 bits FIFO mode. To clear the FIFO's counter and content, it has to be set zero
156  //
157  // NOTE : Synchron word is 2DD4h
158  // NOTE : To restart the synchron word reception, bit ff should be cleared and set. This action will initialize the FIFO and clear its content.
159  // NOTE : Bit fe modifies the function of pin 3 and pin 4. Pin 3 (nFFS) will become input if fe is set to 1. If the chip is used in FIFO mode, do not allow this to be a floating input.
160  SPI_SendWord(0b1100111010001000);
161  DELAI_US(250);
162  SPI_SendWord(0b1100111010001011);
163  DELAI_US(250);
164

166  // ----------

168  // dr : Disable the higly sensitive RESET mode. If this bit is cleared, a 600mV glitch in the power supply may cause a system reset.
169  SPI_SendWord(0b1101101000000001);

293 void ALPHA_SendFSK(uint8_t data)
294 {
295  for (uint8_t offset = 7; offset >= 0; offset--)
296  {
297  while ((PINB & _BV(ALPHA_NIRQ_PIN)) == 0); // Wait nIRQ Hi
298  while ((PINB & _BV(ALPHA_NIRQ_PIN)) != 0); // Wait nIRQ Lo
299

301  {
302  ALPHA_FFS_PORT |= _BV(ALPHA_FFS_PIN);

306  ALPHA_FFS_PORT &= ~_BV(ALPHA_FFS_PIN);
